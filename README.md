# 下载安装
* UBUNTU 
````
sudo apt-get update
sudo apt-get install mysql-server
# 查看 mysql 服务是否运行
systemctl status mysql.service
# 启动 mysql 服务
sudo systemctl mysql start
# 查看 mysql 版本
mysqladmin -u root -p password version
````
